def get_courses
  course_array = []
  courses&.split(' ')&.each do |semcrn|
    Course.where(semcrn: semcrn).each do |course|
      course_array << course
    end
  end
  course_array.sort
end

def to_csv
  courses = get_courses
  CSV.generate do |csv|
    csv << Course.csv_headers
    courses.each do |c|
      csv << Course.to_csv_row(c)
    end
  end
end
