def set_cart
  if current_user
    unless current_user.cart
            current_user.create_cart
            current_user.save
    end
    return current_user.cart
  else
    if cookies[:cart]
            cart = Cart.where(cartid: cookies[:cart])
            if cart.size > 0
              return cart[0]
            else
              cookies.delete :cart
              return nil
            end
    end
  end
end
