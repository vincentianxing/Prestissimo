def to_csv(enroll)
  CSV.generate do |csv|
    csv << Course.csv_headers
    courses.sort.each do |c|
      csv << Course.to_csv_row(c) unless (enroll.zero? && c.enroll.zero?) || (c.status == 'cancelled')
    end
  end
end
