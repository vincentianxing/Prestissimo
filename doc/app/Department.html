def to_csv( enroll )
  qs = '( dept LIKE '+ActiveRecord::Base.connection.quote("%"+dept+"%") + ' AND status!="cancelled"'
  qs << ' AND ( enroll BETWEEN 1 AND 999 )' if (enroll == 0)
  qs << ' )'
  courses = Course.where(qs).sort
  CSV.generate do |csv|
    csv << Course.csv_headers 
    courses.each do |c|
      csv << Course.to_csv_row(c)
    end
  end
end
