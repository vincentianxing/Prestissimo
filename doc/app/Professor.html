def to_csv(enroll)
  CSV.generate do |csv|
    csv << Course.csv_headers 
    self.courses.sort.each do |c|
      csv << Course.to_csv_row(c) unless ((enroll == 0) && (c.enroll == 0)) || (c.status == "cancelled")
    end
  end
end

def toggle_status
  if (self.status == "dark")
    self.status = "active"
  else
    self.status = "dark"
  end
end
