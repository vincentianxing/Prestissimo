def copy_calendar
  Dir[File.join(JAVASCRIPTS, '*.js')].each do |file|
    file = File.split(file).last
    copy_file file, "public/javascripts/#{file}"
  end

  Dir[File.join(STYLESHEETS, '*.css')].each do |file|
    file = File.split(file).last
    copy_file file, "public/stylesheets/#{file}"
  end
end
